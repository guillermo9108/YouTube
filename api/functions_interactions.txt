function interact_has_purchased($pdo, $userId, $videoId) {
    // 1. Obtener datos del video y rol del usuario en una sola pasada
    $stmt = $pdo->prepare("
        SELECT v.creatorId, u_creator.role as creatorRole, 
               u_me.role as myRole, u_me.vipExpiry as myVipExpiry
        FROM videos v 
        LEFT JOIN users u_creator ON v.creatorId = u_creator.id 
        CROSS JOIN users u_me ON u_me.id = ?
        WHERE v.id = ?
    ");
    $stmt->execute([$userId, $videoId]);
    $data = $stmt->fetch(PDO::FETCH_ASSOC);
    
    if ($data) {
        // ADMIN tiene acceso total
        if ($data['myRole'] === 'ADMIN') { respond(true, ['hasPurchased' => true]); return; }
        
        // Creador tiene acceso a su propio video
        if ($data['creatorId'] === $userId) { respond(true, ['hasPurchased' => true]); return; }
        
        // PASE VIP: Desbloquea contenido del administrador
        $isVip = $data['myVipExpiry'] && intval($data['myVipExpiry']) > time();
        if ($isVip && $data['creatorRole'] === 'ADMIN') {
            respond(true, ['hasPurchased' => true]); 
            return; 
        }
    }
    
    // 2. Si no es admin ni vip, verificar compra específica
    $stmt = $pdo->prepare("SELECT COUNT(*) FROM transactions WHERE buyerId = ? AND videoId = ? AND type = 'PURCHASE'");
    $stmt->execute([$userId, $videoId]);
    respond(true, ['hasPurchased' => $stmt->fetchColumn() > 0]);
}

function interact_get_transactions($pdo, $userId) {
    // Corregido: Join robusto para targetId y asegurar que las columnas existan
    $stmt = $pdo->prepare("
        SELECT t.*, 
               u1.username as targetName, 
               u2.username as buyerName,
               v.title as videoTitle
        FROM transactions t 
        LEFT JOIN users u1 ON t.targetId = u1.id 
        LEFT JOIN users u2 ON t.buyerId = u2.id
        LEFT JOIN videos v ON t.videoId = v.id
        WHERE t.buyerId = ? OR t.creatorId = ? OR t.targetId = ? 
        ORDER BY t.timestamp DESC
    ");
    $stmt->execute([$userId, $userId, $userId]);
    respond(true, $stmt->fetchAll(PDO::FETCH_ASSOC));
}

function interact_transfer_balance($pdo, $input) {
    $senderId = $input['userId']; $targetUsername = trim($input['targetUsername']); $amount = floatval($input['amount']);
    if ($amount <= 0) respond(false, null, 'Monto inválido');
    $pdo->beginTransaction();
    try {
        $stmt = $pdo->prepare("SELECT id, username FROM users WHERE username = ?");
        $stmt->execute([$targetUsername]); $target = $stmt->fetch(PDO::FETCH_ASSOC);
        if (!$target) throw new Exception("Receptor no encontrado");
        if ($target['id'] === $senderId) throw new Exception("No puedes enviarte a ti mismo");
        $feePct = $pdo->query("SELECT transferFee FROM system_settings LIMIT 1")->fetchColumn() ?: 5;
        $feeAmount = $amount * ($feePct / 100); $netAmount = $amount - $feeAmount;
        $stmt = $pdo->prepare("SELECT balance FROM users WHERE id = ? FOR UPDATE");
        $stmt->execute([$senderId]); if ($stmt->fetchColumn() < $amount) throw new Exception("Saldo insuficiente");
        $pdo->prepare("UPDATE users SET balance = balance - ? WHERE id = ?")->execute([$amount, $senderId]);
        $pdo->prepare("UPDATE users SET balance = balance + ? WHERE id = ?")->execute([$netAmount, $target['id']]);
        $adminId = $pdo->query("SELECT id FROM users WHERE role = 'ADMIN' LIMIT 1")->fetchColumn();
        if ($adminId) { $pdo->prepare("UPDATE users SET balance = balance + ? WHERE id = ?")->execute([$feeAmount, $adminId]); }
        $pdo->prepare("INSERT INTO transactions (id, buyerId, targetId, amount, adminFee, timestamp, type) VALUES (?, ?, ?, ?, ?, ?, 'TRANSFER_SENT')")->execute([uniqid('tr_'), $senderId, $target['id'], $amount, $feeAmount, time()]);
        $pdo->prepare("INSERT INTO transactions (id, buyerId, targetId, amount, timestamp, type) VALUES (?, ?, ?, ?, ?, 'TRANSFER_RECV')")->execute([uniqid('tr_'), $target['id'], $senderId, $netAmount, time()]);
        $pdo->commit(); respond(true, ['netAmount' => $netAmount, 'fee' => $feeAmount]);
    } catch (Exception $e) { $pdo->rollBack(); respond(false, null, $e->getMessage()); }
}
