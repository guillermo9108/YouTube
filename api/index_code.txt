<?php
/**
 * StreamPay - Core Controller V7.0
 * Protección extrema contra 'Respuesta Corrupta' y restauración de módulos.
 */

// 1. Silencio total de errores directos al output
ini_set('display_errors', 0);
ini_set('log_errors', 1);
error_reporting(E_ALL);

// 2. Iniciar captura de salida inmediatamente
if (ob_get_level() == 0) ob_start();

// Capturar warnings para depuración silenciosa
$GLOBALS['php_warnings'] = [];
set_error_handler(function($errno, $errstr, $errfile, $errline) {
    if (!(error_reporting() & $errno)) return false;
    $GLOBALS['php_warnings'][] = "[$errno] $errstr en $errfile:$errline";
    return true;
});

// Manejador de crash total
register_shutdown_function(function() {
    $error = error_get_last();
    if ($error && in_array($error['type'], [E_ERROR, E_PARSE, E_CORE_ERROR, E_COMPILE_ERROR])) {
        // Limpieza absoluta de cualquier salida previa que rompa el JSON
        while (ob_get_level() > 0) ob_end_clean();
        header('Content-Type: application/json; charset=utf-8');
        echo json_encode([
            'success' => false, 
            'error' => 'CRASH SERVIDOR: ' . $error['message']
        ]);
        exit;
    }
});

/**
 * Única vía de salida del servidor. 
 * Garantiza que el JSON esté limpio de basura (warnings, notices, etc).
 */
function respond($success, $data = null, $error = null) {
    // ELIMINAR CUALQUIER BASURA DEL BÚFER (Crucial para NAS)
    while (ob_get_level() > 0) ob_end_clean();
    
    header('Content-Type: application/json; charset=utf-8');
    
    $res = [
        'success' => $success, 
        'data' => $data, 
        'error' => $error
    ];
    
    // Adjuntar advertencias solo si existen y no estamos en producción estricta
    if (!empty($GLOBALS['php_warnings'])) {
        $res['warnings'] = $GLOBALS['php_warnings'];
    }
    
    $output = json_encode($res, JSON_UNESCAPED_UNICODE | JSON_PARTIAL_OUTPUT_ON_ERROR);
    echo $output;
    exit();
}

header('Access-Control-Allow-Origin: *');
header('Access-Control-Allow-Methods: POST, GET, OPTIONS');
header('Access-Control-Allow-Headers: Content-Type, Authorization');

if ($_SERVER['REQUEST_METHOD'] === 'OPTIONS') { 
    while (ob_get_level() > 0) ob_end_clean();
    exit(); 
}

define('AVATAR_DIR', 'uploads/avatars/');
define('MARKET_DIR', 'uploads/market/');

$action = $_GET['action'] ?? null;
$input = json_decode(file_get_contents('php://input'), true) ?? $_POST;
$configFile = 'db_config.json';

// Carga modular de funciones (Aseguramos que todos los archivos PHP existan)
$required_files = ['functions_schema.php', 'functions_utils.php', 'functions_auth.php', 'functions_videos.php', 'functions_interactions.php', 'functions_market.php', 'functions_admin.php', 'functions_payment.php'];
foreach ($required_files as $file) { 
    if (file_exists($file)) { @require_once $file; }
}

if ($action === 'check_installation') {
    respond(true, ['status' => file_exists($configFile) ? 'installed' : 'not_installed']);
}

if (!file_exists($configFile)) respond(false, null, 'No instalado');
$db_config = json_decode(file_get_contents($configFile), true);

try {
    $pdo = new PDO("mysql:host={$db_config['host']};port={$db_config['port']};dbname={$db_config['name']};charset=utf8mb4", $db_config['user'], $db_config['password'], [
        PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
        PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
        PDO::ATTR_EMULATE_PREPARES => false
    ]);
} catch (PDOException $e) { 
    respond(false, null, 'Error DB: ' . $e->getMessage()); 
}

/**
 * ROUTER PRINCIPAL - Restaurado y Completo
 */
try {

        // SOCIAL & COMENTARIOS

        case 'toggle_subscribe': interact_toggle_subscribe($pdo, $input); break;

        case 'get_subscriptions': interact_get_subscriptions($pdo, $_GET['userId']); break;

    }
} catch (Exception $e) { 
    respond(false, null, $e->getMessage()); 
}
